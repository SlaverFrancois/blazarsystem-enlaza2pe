 function doPost(e) {

  try {

    const sheet =
      SpreadsheetApp
      .getActiveSpreadsheet()
      .getActiveSheet();

    const data =
      JSON.parse(e.postData.contents);

    /* VALIDAR */

    if (!data.servicios ||
    !data.nombretelefono) {

      return respuesta("error");
    }

    /* HONEYPOT */

    if (data.website) {
      return respuesta("spam");
    }

    /* LONGITUD */

    if (
      data.servicios.length > 350 ||
      data.nombretelefono.length > 80
    ) {
      return respuesta("muy largo");
    }

    /* LIMPIAR */

    const servicios =
    sanitizar(data.servicios);

    const nombretelefono =
      sanitizar(data.nombretelefono);

    /* EVITAR FORMULAS */

    const serviciosSeguro =
      evitarFormula(servicios);

    const nombreSeguro =
      evitarFormula(nombretelefono);

    /* RATE LIMIT SIMPLE */

    const cache =
      CacheService.getScriptCache();

    const ipKey =
      "spam_" + nombreSeguro;

    const existe =
      cache.get(ipKey);

    if (existe) {
      return respuesta("espera");
    }

    cache.put(ipKey, "1", 60);

    /* GUARDAR */

    sheet.appendRow([
      serviciosSeguro,
      nombreSeguro,
      new Date()
    ]);

    return respuesta("ok");

  } catch(err) {

    return respuesta("error");

  }

}

/* SANITIZAR */

function sanitizar(texto) {

  if (!texto) return "";

  return texto
    .replace(/[<>]/g, "")
    .replace(/script/gi, "")
    .replace(/javascript:/gi, "")
    .replace(/onerror/gi, "")
    .replace(/onload/gi, "")
    .trim();

}

/* EVITAR FORMULAS */

function evitarFormula(texto) {

  if (
    texto.startsWith("=") ||
    texto.startsWith("+") ||
    texto.startsWith("-") ||
    texto.startsWith("@")
  ) {
    return "'" + texto;
  }

  return texto;

}

/* RESPUESTA */

function respuesta(msg) {

  return ContentService
    .createTextOutput(
      JSON.stringify({
        result: msg
      })
    )
    .setMimeType(
      ContentService.MimeType.JSON
    );

}
